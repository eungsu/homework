		반환하는 정적 메소드 정의하기

		    static int[] randomInteger() {
			int[] numbers = new int[5]
			
			// 1부터 100까지의 무작위 숫자 획득 : int x = (int)(Math.random()*100) + 1
			// Math.random()과 for문(5회반복)을 사용해서 임의의 난수를 
			// 뽑아서 배열에 순서대로 저장하는 코드 작성

			return numbers;
		    }
		}
